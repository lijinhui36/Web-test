	calss Animal{
		//构造器
		constructor(name,age){
			//实例属性 小animal的属性
			this.name=name;
			this.age=age;
		}
		//实例的方法,小animal方法
		sayName(){
			console.log(this.name);
		}
		//static 类方法 静态方法
		static say(){
			console.log('say');
		}
	}
	Animal.staticAttr();
	Animal.say();
